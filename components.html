import React, { useState } from 'react';

const Input = ({
  type = 'text',
  placeholder,
  value,
  disabled = false,
  error = false,
  onChange
}) => {
  const [internalValue, setInternalValue] = useState(value || '');

  const handleChange = (e) => {
    const newValue = e.target.value;
    setInternalValue(newValue);
    onChange?.(newValue, e);
  };

  return (
    <input
      type={type}
      className={`input ${error ? 'error' : ''}`}
      placeholder={placeholder}
      value={value !== undefined ? value : internalValue}
      disabled={disabled}
      onChange={handleChange}
    />
  );
};

export default Input;

import React, { useEffect } from 'react';
import { createPortal } from 'react-dom';

const Modal = ({ 
  isOpen, 
  onClose, 
  title, 
  children,
  closeOnOverlayClick = true 
}) => {
  useEffect(() => {
    const handleEscape = (e) => {
      if (e.key === 'Escape') onClose();
    };

    if (isOpen) {
      document.addEventListener('keydown', handleEscape);
      document.body.style.overflow = 'hidden';
    }

    return () => {
      document.removeEventListener('keydown', handleEscape);
      document.body.style.overflow = 'unset';
    };
  }, [isOpen, onClose]);

  if (!isOpen) return null;

  return createPortal(
    <div className="modal-overlay" onClick={closeOnOverlayClick ? onClose : undefined}>
      <div className="modal-content" onClick={e => e.stopPropagation()}>
        {title && <h2>{title}</h2>}
        <div className="modal-body">{children}</div>
        <button className="modal-close" onClick={onClose}>√ó</button>
      </div>
    </div>,
    document.body
  );
};

export default Modal;

class Modal {
  constructor(selector, options = {}) {
    this.modal = document.querySelector(selector);
    this.options = {
      closeOnOverlayClick: true,
      closeOnEsc: true,
      ...options
    };
    this.init();
  }

  init() {
    if (!this.modal) return;

    if (this.options.closeOnOverlayClick) {
      this.modal.addEventListener('click', (e) => {
        if (e.target === this.modal) this.close();
      });
    }

    if (this.options.closeOnEsc) {
      document.addEventListener('keydown', (e) => {
        if (e.key === 'Escape' && this.isOpen()) this.close();
      });
    }
  }

  open() {
    this.modal.style.display = 'flex';
    document.body.style.overflow = 'hidden';
  }

  close() {
    this.modal.style.display = 'none';
    document.body.style.overflow = 'auto';
  }

  isOpen() {
    return this.modal.style.display === 'flex';
  }
}

class Tabs {
  constructor(container) {
    this.container = container;
    this.tabs = container.querySelectorAll('.tab');
    this.contents = container.querySelectorAll('.tab-content');
    this.init();
  }

  init() {
    this.tabs.forEach(tab => {
      tab.addEventListener('click', () => {
        const tabId = tab.dataset.tab;
        this.activateTab(tabId);
      });
    });
  }

  activateTab(tabId) {
    this.tabs.forEach(t => t.classList.remove('active'));
    this.contents.forEach(c => c.classList.remove('active'));

    const targetTab = this.container.querySelector(`[data-tab="${tabId}"]`);
    const targetContent = this.container.querySelector(`#${tabId}`);

    if (targetTab) targetTab.classList.add('active');
    if (targetContent) targetContent.classList.add('active');
  }
}

document.addEventListener('DOMContentLoaded', () => {
  document.querySelectorAll('.tabs-container').forEach(container => {
    new Tabs(container);
  });
});

class Accordion {
  constructor(container) {
    this.container = container;
    this.items = container.querySelectorAll('.accordion-item');
    this.init();
  }

  init() {
    this.items.forEach(item => {
      const header = item.querySelector('.accordion-header');
      const content = item.querySelector('.accordion-content');
      const icon = header.querySelector('.accordion-icon');

      header.addEventListener('click', () => {
        const isOpen = content.style.maxHeight;
        this.closeAll();
        
        if (!isOpen) {
          content.style.maxHeight = content.scrollHeight + 'px';
          header.classList.add('active');
          icon.textContent = '‚àí';
        }
      });
    });
  }

  closeAll() {
    this.items.forEach(item => {
      const header = item.querySelector('.accordion-header');
      const content = item.querySelector('.accordion-content');
      const icon = header.querySelector('.accordion-icon');

      content.style.maxHeight = null;
      header.classList.remove('active');
      icon.textContent = '+';
    });
  }
}

document.addEventListener('DOMContentLoaded', () => {
  document.querySelectorAll('.accordion').forEach(acc => {
    new Accordion(acc);
  });
});
