import React from 'react';

export const Button = ({ 
  children, 
  variant = 'primary',
  size = 'md',
  disabled = false,
  loading = false,
  onClick 
}) => {
  const baseClasses = 'btn';
  const variantClass = `btn-${variant}`;
  const sizeClass = `btn-${size}`;
  const loadingClass = loading ? 'loading' : '';
  
  return (
    <button
      className={`${baseClasses} ${variantClass} ${sizeClass} ${loadingClass}`}
      disabled={disabled || loading}
      onClick={onClick}
    >
      {loading && <div className="spinner"></div>}
      {children}
    </button>
  );
};
// Input component
import React, { useState } from 'react';

export const Input = ({
  type = 'text',
  placeholder,
  value,
  disabled = false,
  error = false,
  onChange
}) => {
  const [internalValue, setInternalValue] = useState(value || '');

  const handleChange = (e) => {
    const newValue = e.target.value;
    setInternalValue(newValue);
    onChange?.(newValue, e);
  };

  return (
    <input
      type={type}
      className={`input ${error ? 'error' : ''} ${disabled ? 'disabled' : ''}`}
      placeholder={placeholder}
      value={internalValue}
      disabled={disabled}
      onChange={handleChange}
    />
  );
};
// Modal component
import React, { useEffect } from 'react';
import { createPortal } from 'react-dom';

export const Modal = ({ 
  isOpen, 
  onClose, 
  title, 
  children,
  closeOnOverlayClick = true 
}) => {
  useEffect(() => {
    const handleEscape = (e) => {
      if (e.key === 'Escape') onClose();
    };

    if (isOpen) {
      document.addEventListener('keydown', handleEscape);
      document.body.style.overflow = 'hidden';
    }

    return () => {
      document.removeEventListener('keydown', handleEscape);
      document.body.style.overflow = 'unset';
    };
  }, [isOpen, onClose]);

  if (!isOpen) return null;

  return createPortal(
    <div className="modal-overlay" onClick={closeOnOverlayClick ? onClose : undefined}>
      <div className="modal-content" onClick={e => e.stopPropagation()}>
        {title && <h2>{title}</h2>}
        <div className="modal-body">{children}</div>
        <button className="modal-close" onClick={onClose}>√ó</button>
      </div>
    </div>,
    document.body
  );
};

// Modal class
class Modal {
  constructor(selector, options = {}) {
    this.modal = document.querySelector(selector);
    this.options = {
      closeOnOverlayClick: true,
      closeOnEsc: true,
      ...options
    };
    this.init();
  }

  init() {
    if (!this.modal) return;

    // Overlay click
    if (this.options.closeOnOverlayClick) {
      this.modal.addEventListener('click', (e) => {
        if (e.target === this.modal) this.close();
      });
    }

    // ESC key
    if (this.options.closeOnEsc) {
      document.addEventListener('keydown', (e) => {
        if (e.key === 'Escape' && this.isOpen()) this.close();
      });
    }
  }

  open() {
    this.modal.style.display = 'flex';
    document.body.style.overflow = 'hidden';
  }

  close() {
    this.modal.style.display = 'none';
    document.body.style.overflow = 'auto';
  }

  isOpen() {
    return this.modal.style.display === 'flex';
  }
}

// Tabs class
class Tabs {
  constructor(container) {
    this.container = container;
    this.tabs = container.querySelectorAll('.tab');
    this.contents = container.querySelectorAll('.tab-content');
    this.init();
  }

  init() {
    this.tabs.forEach(tab => {
      tab.addEventListener('click', () => {
        const tabId = tab.dataset.tab;
        this.activateTab(tabId);
      });
    });
  }

  activateTab(tabId) {
    // Deactivate all
    this.tabs.forEach(t => t.classList.remove('active'));
    this.contents.forEach(c => c.classList.remove('active'));

    // Activate target
    const targetTab = this.container.querySelector(`[data-tab="${tabId}"]`);
    const targetContent = this.container.querySelector(`#${tabId}`);

    if (targetTab) targetTab.classList.add('active');
    if (targetContent) targetContent.classList.add('active');
  }
}

// Auto-initialize
document.addEventListener('DOMContentLoaded', () => {
  document.querySelectorAll('.tabs-container').forEach(container => {
    new Tabs(container);
  });
});

// Accordion class
class Accordion {
  constructor(container) {
    this.container = container;
    this.items = container.querySelectorAll('.accordion-item');
    this.init();
  }

  init() {
    this.items.forEach(item => {
      const header = item.querySelector('.accordion-header');
      const content = item.querySelector('.accordion-content');
      const icon = header.querySelector('.accordion-icon');

      header.addEventListener('click', () => {
        const isOpen = content.style.maxHeight;

        // Close all items
        this.closeAll();

        // Open current if it was closed
        if (!isOpen) {
          content.style.maxHeight = content.scrollHeight + 'px';
          header.classList.add('active');
          icon.textContent = '‚àí';
        }
      });
    });
  }

  closeAll() {
    this.items.forEach(item => {
      const header = item.querySelector('.accordion-header');
      const content = item.querySelector('.accordion-content');
      const icon = header.querySelector('.accordion-icon');

      content.style.maxHeight = null;
      header.classList.remove('active');
      icon.textContent = '+';
    });
  }

  openItem(index) {
    const item = this.items[index];
    if (item) {
      item.querySelector('.accordion-header').click();
    }
  }
}

class Carousel {
  constructor(container, options = {}) {
    this.container = container;
    this.slides = container.querySelectorAll('.carousel-slide');
    this.currentIndex = 0;
    this.options = {
      autoPlay: false,
      interval: 5000,
      ...options
    };
    this.init();
  }

  init() {
    this.createControls();
    this.showSlide(this.currentIndex);

    if (this.options.autoPlay) {
      this.startAutoPlay();
    }
  }

  createControls() {
    const prevBtn = document.createElement('button');
    const nextBtn = document.createElement('button');

    prevBtn.className = 'carousel-prev';
    nextBtn.className = 'carousel-next';

    prevBtn.innerHTML = '‚Äπ';
    nextBtn.innerHTML = '‚Ä∫';

    prevBtn.addEventListener('click', () => this.prev());
    nextBtn.addEventListener('click', () => this.next());

    this.container.appendChild(prevBtn);
    this.container.appendChild(nextBtn);
  }

  showSlide(index) {
    this.slides.forEach((slide, i) => {
      slide.style.display = i === index ? 'block' : 'none';
    });
    this.currentIndex = index;
  }

  next() {
    let newIndex = this.currentIndex + 1;
    if (newIndex >= this.slides.length) newIndex = 0;
    this.showSlide(newIndex);
  }

  prev() {
    let newIndex = this.currentIndex - 1;
    if (newIndex < 0) newIndex = this.slides.length - 1;
    this.showSlide(newIndex);
  }

  startAutoPlay() {
    this.autoPlayInterval = setInterval(() => {
      this.next();
    }, this.options.interval);
  }

  stopAutoPlay() {
    if (this.autoPlayInterval) {
      clearInterval(this.autoPlayInterval);
    }
  }
}

// FormValidator class
class FormValidator {
  constructor(form, rules) {
    this.form = form;
    this.rules = rules;
    this.errors = {};
    this.init();
  }

  init() {
    this.form.addEventListener('submit', (e) => {
      if (!this.validate()) {
        e.preventDefault();
        this.showErrors();
      }
    });

    // Real-time validation
    Object.keys(this.rules).forEach(fieldName => {
      const field = this.form.querySelector(`[name="${fieldName}"]`);
      if (field) {
        field.addEventListener('blur', () => this.validateField(fieldName));
        field.addEventListener('input', () => this.clearFieldError(fieldName));
      }
    });
  }

  validate() {
    this.errors = {};
    
    Object.keys(this.rules).forEach(fieldName => {
      this.validateField(fieldName);
    });

    return Object.keys(this.errors).length === 0;
  }

  validateField(fieldName) {
    const field = this.form.querySelector(`[name="${fieldName}"]`);
    const value = field?.value.trim();
    const rules = this.rules[fieldName];

    for (let rule of rules) {
      const error = this.checkRule(rule, value, fieldName);
      if (error) {
        this.errors[fieldName] = error;
        break;
      }
    }

    if (!this.errors[fieldName]) {
      this.clearFieldError(fieldName);
    }

    return !this.errors[fieldName];
  }

  checkRule(rule, value, fieldName) {
    switch (rule.type) {
      case 'required':
        if (!value) return `${fieldName} is required`;
        break;
      case 'email':
        if (value && !/^[^\s@]+@[^\s@]+\.[^\s@]+$/.test(value)) {
          return 'Invalid email format';
        }
        break;
      case 'minLength':
        if (value && value.length < rule.value) {
          return `Minimum ${rule.value} characters required`;
        }
        break;
    }
    return null;
  }

  showErrors() {
    this.clearErrors();
    
    Object.keys(this.errors).forEach(fieldName => {
      const field = this.form.querySelector(`[name="${fieldName}"]`);
      const errorElement = document.createElement('div');
      errorElement.className = 'field-error';
      errorElement.textContent = this.errors[fieldName];
      
      field.parentNode.appendChild(errorElement);
      field.classList.add('error');
    });
  }

  clearErrors() {
    this.form.querySelectorAll('.field-error').forEach(el => el.remove());
    this.form.querySelectorAll('.error').forEach(el => el.classList.remove('error'));
  }

  clearFieldError(fieldName) {
    const field = this.form.querySelector(`[name="${fieldName}"]`);
    if (field) {
      field.classList.remove('error');
      const errorElement = field.parentNode.querySelector('.field-error');
      if (errorElement) errorElement.remove();
    }
    delete this.errors[fieldName];
  }
}

    { type: 'required' },
    { type: 'email' }
  ],
  password: [
    { type: 'required' },
    { type: 'minLength', value: 6 }
  ]
});
