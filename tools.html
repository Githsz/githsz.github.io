Auto Commit

–ê–≤—Ç–æ–º–∞—Ç–∏—á–µ—Å–∫–æ–µ —Å–æ–∑–¥–∞–Ω–∏–µ –∫–æ–º–º–∏—Ç–æ–≤ —Å —É–º–Ω—ã–º–∏ —Å–æ–æ–±—â–µ–Ω–∏—è–º–∏

#!/bin/bash
# git-auto-commit.sh
# –ê–≤—Ç–æ–º–∞—Ç–∏—á–µ—Å–∫–æ–µ —Å–æ–∑–¥–∞–Ω–∏–µ –∫–æ–º–º–∏—Ç–∞ —Å –∞–Ω–∞–ª–∏–∑–æ–º –∏–∑–º–µ–Ω–µ–Ω–∏–π

generate_commit_message() {
    local added_files=$(git diff --cached --name-status | grep -E '^A' | wc -l)
    local modified_files=$(git diff --cached --name-status | grep -E '^M' | wc -l)
    local deleted_files=$(git diff --cached --name-status | grep -E '^D' | wc -l)
    
    if [ $added_files -gt 0 ]; then
        echo "feat: add $added_files new file(s)"
    elif [ $modified_files -gt 0 ]; then
        echo "fix: update $modified_files file(s)"
    elif [ $deleted_files -gt 0 ]; then
        echo "refactor: remove $deleted_files file(s)"
    else
        echo "chore: update repository"
    fi
}

COMMIT_MSG=$(generate_commit_message)
git commit -m "$COMMIT_MSG"
